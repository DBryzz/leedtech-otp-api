import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

interface FeePayment {
    studentNumber: string;
    paymentAmount: number;
    paymentDate: string;
    incentiveRate: number;
    incentiveAmount: number;
    newBalance: number;
    nextPaymentDueDate: string;
}

@Injectable({
    providedIn: 'root',
})
export class FeePaymentService {
    private apiUrl = 'http://localhost:8080/one-time-fee-payment';

    constructor(private http: HttpClient) {}

    makePayment(studentNumber: string, paymentAmount: number, paymentDate?: string): Observable<FeePayment> {
        return this.http.post<FeePayment>(this.apiUrl, { studentNumber, paymentAmount, paymentDate });
    }
}
