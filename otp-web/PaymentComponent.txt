import { Component } from '@angular/core';
import { FeePaymentService } from './fee-payment.service';

@Component({
    selector: 'app-payment',
    template: `
    <form (ngSubmit)="onSubmit()">
        <input [(ngModel)]="studentNumber" name="studentNumber" placeholder="Student Number" required>
        <input [(ngModel)]="paymentAmount" name="paymentAmount" placeholder="Payment Amount" required>
        <input [(ngModel)]="paymentDate" name="paymentDate" placeholder="Payment Date (Optional)" type="date">
        <button type="submit">Submit Payment</button>
    </form>
    <div *ngIf="paymentResult">
        <p>New Balance: {{paymentResult.newBalance}}</p>
        <p>Next Payment Due Date: {{paymentResult.nextPaymentDueDate}}</p>
    </div>
    `
})
export class PaymentComponent {
    studentNumber: string;
    paymentAmount: number;
    paymentDate: string;
    paymentResult: any;

    constructor(private feePaymentService: FeePaymentService) {}

    onSubmit() {
        this.feePaymentService.makePayment(this.studentNumber, this.paymentAmount, this.paymentDate)
            .subscribe(result => this.paymentResult = result);
    }
}
